JavaScript/React Example

App.jsx
import React, { useState, useEffect } from 'react'
import { Chart } from 'react-chartjs-2'
const DataVisualization = () => {
const [data, setData] = useState(null)
const [loading, setLoading] = useState(true)
useEffect(() => {
fetchData()
}, [])
const fetchData = async () => {
try {
const response = await fetch('/api/data')
const result = await response.json()
setData(result)
} catch (error) {
console.error('Error fetching data:', error)
} finally {
setLoading(false)
}
}
if (loading) return <div>Loading...div>
return (
<div className="chart-container">
<Chart data={data} options={{ responsive: true }} />
div>
)
}
export default DataVisualization

R Data Analysis Example

analysis.R
# Water Quality Analysis in England
# Loading required libraries
library("tidyverse")
library("ggplot2")
library("dplyr")
# Read the dataset
water_data <- read_csv("water_quality_england.csv")
# Data preprocessing
cleaned_data <- water_data %>%
filter(!is.na(pollution_level)) %>%
mutate(
year = year(date),
severity_category = case_when(
pollution_level < 5 ~ "Low",
pollution_level <= 10 ~ "Medium",
TRUE ~ "High"
)
)
# Create visualization
pollution_trend <- ggplot(cleaned_data, aes(x = year, y = pollution_level)) +
geom_line(aes(color = severity_category)) +
labs(
title = "Water Pollution Trends in England (2000-2023)",
x = "Year",
y = "Pollution Level",
color = "Severity"
) +
theme_minimal()
# Display the plot
print(pollution_trend)
